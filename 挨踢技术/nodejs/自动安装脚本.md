# 自动安装脚本

```
#!/bin/bash
 
tar zxf $1.tar.gz
 
sudo rm -rf /opt/$1
sudo cp -r ./$1 /opt

sudo rm -rf /usr/bin/node
sudo rm -rf /usr/bin/npx
sudo rm -rf /usr/bin/npm
sudo rm -rf /usr/bin/ng
sudo rm -rf /usr/bin/cnpm
sudo rm -rf /usr/bin/yarn
 
sudo ln -s /opt/$1/bin/node /usr/bin/
sudo ln -s /opt/$1/bin/npm /usr/bin/
sudo ln -s /opt/$1/bin/npx /usr/bin/
 
cd ~

sudo npm install -g cnpm
sudo npm install -g yarn
sudo ln -s /opt/$1/bin/cnpm /usr/bin/
sudo ln -s /opt/$1/bin/yarn /usr/bin/
```
