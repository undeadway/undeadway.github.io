# 上传文件重命名脚本

```
#!/bin/bash

########################################
#     上传文件重命名脚本               #
#     作者    ：waygc                  #
#     更新时间：2023年12月19日         #
########################################

dat=`date +"%Y%m%d-%H%M%S"`
dat1=`date +"%Y%m"`
if [ ! -d "./$dat1/" ]; then
  mkdir "$dat1"
fi
mv dist/ "$dat1/$dat"
```
